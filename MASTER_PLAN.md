### Database Schema

```sql
-- Users (you and partner)
CREATE TABLE users (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  role TEXT DEFAULT 'member', -- 'admin' | 'member'
  avatar_url TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Shared Checklist
CREATE TABLE tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  description TEXT,
  category TEXT NOT NULL, -- 'visa' | 'income' | 'life' | 'tech'
  priority TEXT DEFAULT 'medium', -- 'critical' | 'high' | 'medium' | 'low'
  status TEXT DEFAULT 'todo', -- 'todo' | 'in_progress' | 'done' | 'blocked'
  due_date DATE,
  assigned_to UUID REFERENCES users(id),
  created_by UUID REFERENCES users(id),
  depends_on UUID REFERENCES tasks(id),
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Income Tracking
CREATE TABLE income_entries (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  source TEXT NOT NULL, -- 'dataannotation' | 'freelance' | 'saas' | 'digital_products' | 'other'
  amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'USD', -- 'USD' | 'JPY' | 'MYR'
  layer INTEGER NOT NULL, -- 1, 2, or 3
  description TEXT,
  date DATE NOT NULL,
  user_id UUID REFERENCES users(id),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Monthly income goals
CREATE TABLE income_goals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  month DATE NOT NULL, -- first day of month
  target_amount DECIMAL(10,2) NOT NULL,
  currency TEXT DEFAULT 'USD',
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Visa Progress
CREATE TABLE visa_steps (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  description TEXT,
  status TEXT DEFAULT 'pending', -- 'pending' | 'in_progress' | 'completed' | 'blocked'
  due_date DATE,
  completed_at TIMESTAMPTZ,
  notes TEXT,
  sort_order INTEGER NOT NULL,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Documents
CREATE TABLE documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  category TEXT NOT NULL, -- 'visa' | 'financial' | 'personal' | 'business'
  file_url TEXT NOT NULL,
  file_type TEXT, -- 'pdf' | 'image' | 'doc'
  uploaded_by UUID REFERENCES users(id),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- Decision Log
CREATE TABLE decisions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  context TEXT, -- why this decision came up
  options TEXT, -- JSON array of considered options
  chosen TEXT NOT NULL, -- what was decided
  reasoning TEXT, -- why this option was chosen
  decided_by UUID REFERENCES users(id),
  decided_at TIMESTAMPTZ DEFAULT now(),
  category TEXT -- 'visa' | 'income' | 'life' | 'tech'
);

-- Research Notes
CREATE TABLE notes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  content TEXT NOT NULL, -- markdown
  category TEXT, -- 'visa' | 'housing' | 'income' | 'lifestyle' | 'tech'
  tags TEXT[], -- array of tags
  created_by UUID REFERENCES users(id),
  pinned BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

-- Goals
CREATE TABLE goals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  description TEXT,
  category TEXT NOT NULL, -- 'financial' | 'visa' | 'business' | 'personal'
  target_value DECIMAL(10,2), -- e.g., savings target
  current_value DECIMAL(10,2) DEFAULT 0,
  unit TEXT, -- 'USD' | 'JPY' | 'count' | 'percent'
  deadline DATE,
  status TEXT DEFAULT 'active', -- 'active' | 'completed' | 'paused'
  created_at TIMESTAMPTZ DEFAULT now()
);
```
